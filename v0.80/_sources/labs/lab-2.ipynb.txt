    "@experiments\n",
    "def LE(cfg: pax.Config) -> pax.Config:\n",
    "    cfg.carriers.AL1.rm_system = \"L\"\n",
    "    cfg.carriers.AL1.frat5 = \"curve_C\"  # curve_A through curve_G are available\n",
    "    cfg.carriers.AL1.rm_system_options = dict(\n",
    "        # fare_adjustment=\"ki\",           # \"ki\" or \"mr\" or None\n",
    "        # fare_adjustment_scale=0.5,      # between 0.0 and 1.0\n",
    "        # regression_weight=\"sellup\",     # \"sellup\", \"sellup^2\", \"fare\", \"none\"\n",
    "        # variance_rollup_algorithm=\"tf\", # \"tf\" or \"dep\"\n",
    "        # variance_is_ratio_of_mean=0.0,  # 0.0 or some positive float, typically 2.0\n",
    "        # max_cap=0.0,                    # 0.0 or some positive float, typically 10.0\n",
    "        # q_allocation_algorithm=\"tf\",    # \"tf\" or \"dep\"\n",

    "from IPython.display import display\n",
    "\n",
    "\n",
    "def try_me(cfg: pax.Config, against: str = \"Baseline\") -> None:\n",
    "    cfg = cfg.model_copy(deep=True)\n",
    "    cfg.carriers.AL1.rm_system = \"M\"\n",
    "    cfg.carriers.AL1.frat5 = \"curve_B\"  # curve_A through curve_G are available\n",
    "    cfg.carriers.AL1.rm_system_options = dict(\n",
    "        # fare_adjustment=\"mr\",           # \"ki\" or \"mr\" or None\n",
    "        # bid_price_vector=True,          # True or False\n",
    "        # fare_adjustment_scale=.75,      # between 0.0 and 1.0\n",
    "        # regression_weight=\"sellup\",     # \"sellup\", \"sellup^2\", \"fare\", \"none\"\n",
    "        # variance_rollup_algorithm=\"tf\", # \"tf\" or \"dep\"\n",
    "        # variance_is_ratio_of_mean=0.0,  # 0.0 or some positive float, typically 2.0\n",
    "        # max_cap=0.0,                    # 0.0 or some positive float, typically 10.0\n",
    "        # q_allocation_algorithm=\"tf\",    # \"tf\" or \"dep\"\n",
    "    )\n",
    "    sim = pax.MultiSimulation(cfg)\n",
    "    out = sim.run()\n",
    "    comp = Contrast({against: results[against], \"try_me\": out})\n",
    "    display(comp.fig_carrier_revenues())\n",
    "    display(comp.fig_carrier_load_factors())\n",
    "\n",
    "\n",
    "try_me(cfg)"
